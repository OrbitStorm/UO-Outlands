// User setup
headmsg 'Select From Bag' 38
promptalias 'FromBag'
useobject 'FromBag'
pause 700
headmsg 'Select Sell Bag' 38
promptalias 'SellBag'
useobject 'SellBag'
pause 700
headmsg 'Select Keep Bag' 38
promptalias 'KeepBag'
useobject 'KeepBag'
pause 700
// Debug
@clearjournal
@clearignorelist
@unsetalias 'IDme'
// 120 Item ID does whole bag first
if skill 'Item Identification' == 120
  useskill 'Item Identification'
  waitfortarget 1200
  target! 'FromBag'
  pause 1000
endif
for 0 to 'ItemTypes'
  while @findtype 'ItemTypes[]' 'any' 'FromBag'
    @setalias 'IDme' 'found'
    @clearjournal
    if skill 'Item Identification' == 120
      clickobject 'IDme'
    else
      useskill 'Item Identification'
      waitfortarget 5000
      target! 'IDme'
    endif
    // Edit this pause longer if your ping is above 200
    pause 800
    if @injournal "not certain" system
      headmsg "Skill failed, trying again" 38
      pause 200
    else
      // Checks for any items you wish to keep
      for 0 to 'ItemQuality'
        if @injournal 'ItemQuality[]' 'system'
          moveitem 'IDme' 'KeepBag'
          @ignoreobject 'IDme'
          @unsetalias 'IDme'
          headmsg 'ItemQuality[]' 38
        endif
      endfor
      pause 200
      if @findobject 'IDme' 'any' 'FromBag'
        @moveitem 'IDme' 'SellBag'
        @ignoreobject 'IDme'
        @unsetalias 'IDme'
      endif
    endif
    pause 200
  endwhile
endfor
headmsg 'All Done!' 38
